#include <stdio.h>
#include <stdlib.h>
#include <time.h>

int main(void)
{
    char puzzle[10][10] = {
            {'C','P','*','*','Q','*','*','U','*','*'},
            {'*','A','*','*','U','*','*','N','*','*'},
            {'*','N','R','*','A','*','*','I','*','*'},
            {'*','D','*','*','R','*','*','V','*','*'},
            {'*','E','*','*','A','*','*','E','*','*'},
            {'*','M','*','*','N','*','*','R','*','*'},
            {'*','I','*','*','T','*','*','S','*','*'},
            {'*','C','*','*','I','*','*','I','*','*'},
            {'*','*','*','*','N','*','*','T','*','*'},
            {'B','U','S','*','E','*','*','Y','*','*'}
    };

    srand(time(NULL));

    for (int i = 0; i<10; i++){
        for(int j = 0; j< 10; j++){
            if (puzzle[i][j] == '*'){
                puzzle[i][j] = (rand() % 26) + 65;
            }
        }
    }

    for (int i = 0; i < 10; ++i) {
        for (int j = 0; j < 10; ++j) {
            printf("%c  ", puzzle[i][j]);
        }
        printf("\n");
    }

    return 0;
//
}
